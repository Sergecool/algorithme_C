Exercice 2.4
Algorithme prixTTC;
var prixHT,tauxTVA,prixTTC,qte:numeriques;
debut
	ecrire "Entrez le prix HT: ";
	lire prixHT;
	ecrire "Entrez la quantité: ";
	lire qte;
	ecrire "Entrez le taux TVA: ";
	lire tauxTVA;
	prixTTC <-- prixHT * (1 + tauxTVA)
	ecrire "Le prix TTC est: " & prixTTC;
fin.
